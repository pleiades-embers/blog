### 数组扁平化
:::demo 
```vue
<template>
  <div>
    <div>输入数组:{{ arr }}</div>
    <div>Array.concat: {{ flatten1(arr) }}</div>
    <div>扩展运算符: {{ flatten2(arr) }}</div>
  </div>
</template>
<script>
export default {
  name: "",
  data() {
    return {
      arr: [0, 11, [20, 5, [20,[1], 0], 0, 11], 20, 5],
    };
  },
  methods: {
    flatten1(arr) {  //递归
      let result = [];

      arr.forEach((item) => {
        if (Array.isArray(item)) {
          result = result.concat(this.flatten1(item));
        } else {
          result.push(item);
        }
      });
      return result;
    },

    flatten2(arr) { //扩展运算符
        while(arr.some(item=>Array.isArray(item))){
            arr=[].concat(...arr)
        }
        return arr
    },
  },
};
</script>


```
